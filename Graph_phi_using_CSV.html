파이스크립트에서 판다스(Pandas)로 CSV 읽기

import pandas as pd from pyodide.http import open_url df = pd.read_csv(open_url("https://raw.githubusercontent.com/BrightenTheWorld/01_Project/main/df.csv")) phi = pd.read_csv(open_url("https://raw.githubusercontent.com/BrightenTheWorld/01_Project/main/phi.csv")) DataFrame = pd.concat([df,phi],axis=1) csv = Element('csv') csv.write(phi)

읽은 CSV로 그래프 플롯하기

import matplotlib.pyplot as plt import numpy as np plt.rcParams.update({'font.size': 7}) fig = plt.figure(figsize=(10, 5)) plt.plot(df['beta'], phi['phi_1'], label='D=0.01' ) plt.plot(df['beta'], phi['phi_2'], label='D=0.2' ) plt.plot(df['beta'], phi['phi_3'], label='D=0.4' ) plt.plot(df['beta'], phi['phi_4'], label='D=0.707' ) plt.xlim([0, 2]) # X축의 범위: [xmin, xmax] plt.ylim([0, np.pi]) # Y축의 범위: [ymin, ymax] plt.grid(True) pltdiv = Element('pltdiv') pltdiv.write(fig)
