파이스크립트에서 판다스(Pandas)로 CSV 읽기

import pandas as pd from pyodide.http import open_url # URL 에서 csv 데이터를 읽어 판다스 배열로 변환합니다. df = pd.read_csv(open_url("https://raw.githubusercontent.com/BrightenTheWorld/01_Project/main/df.csv")) # 출력할 HTML 요소를 선택 csv = Element('csv') # 해당 HTML 요소에 결과를 출력합니다. csv.write(df) phi = pd.read_csv(open_url("https://raw.githubusercontent.com/BrightenTheWorld/01_Project/main/phi.csv")) csv.write(phi) DataFrame = pd.concat([df,phi],axis=1) csv.write(DataFrame)
